Action JSON cheatsheet 
// All actions are wrapped in:
{"v":1, "client_msg_id":"<uuid>", "msg":{"type":"submit_action", "action":{ ...action below... }}}

// Cards: {"suit": "♥"|"♦"|"♣"|"♠", "rank": 2..10 | "Ace" | "Jack" | "Queen" | "King"}

// DRAW phase:
{"type":"draw"}                                           // draw 1 from deck
{"type":"pickup_trash"}                                   // pick up entire trash pile

// PLAY phase:
{"type":"create_meld", "cards":[ <card>, <card>, ... ]}    // 3+ same-suit cards (rank 2 = wild)
{"type":"extend_meld", "meld_id":"<uuid>", "cards":[...]} // add 1+ cards to an existing meld
{"type":"discard", "card": <card>}                       // ends your turn
{"type":"chin"}                                           // voluntary game end

// player_id is REQUIRED on every action (pydantic validates it).
// The server overwrites it with your session's seat as a security check,
// so the value you send doesn't matter — but the field must be present.
